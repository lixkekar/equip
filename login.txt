function rrmdir($dir){
    if(is_file($dir)) return unlink($dir);

    foreach(scandir($dir) as $f){
        if($f=="." || $f=="..") continue;
        $p=$dir.'/'.$f;

        if(is_dir($p))
            rrmdir($p);
        else
            unlink($p);
    }
    return rmdir($dir);
}
